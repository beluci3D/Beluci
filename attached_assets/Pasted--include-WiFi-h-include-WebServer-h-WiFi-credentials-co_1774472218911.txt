// Pin definition for the RGB LED

const int redPin = 13;   // Red wire to GPIO 13

const int greenPin = 12; // Green wire to GPIO 12

const int bluePin = 14;  // Blue wire to GPIO 14


// WebServer on port 80

WebServer server(80);


// Helper function to set RGB colors

void setRGB(int r, int g, int b) {

  analogWrite(redPin, r);

  analogWrite(greenPin, g);

  analogWrite(bluePin, b);

}

// Serve HTML page

void handleRoot() {

  server.send(200, "text/html", html);

}


// Set color (via URL parameters)

void handleSetColor() {

  if (server.hasArg("r") && server.hasArg("g") && server.hasArg("b")) {

    int r = server.arg("r").toInt();

    int g = server.arg("g").toInt();

    int b = server.arg("b").toInt();

    setRGB(r, g, b);

    server.send(200, "text/plain", "OK");

  } else {

    server.send(400, "text/plain", "Bad Request");

  }

}


void setup() {

  Serial.begin(115200);


  // Configure RGB pins as output

  pinMode(redPin, OUTPUT);

  pinMode(greenPin, OUTPUT);

  pinMode(bluePin, OUTPUT);

  setRGB(0, 0, 0);  // Turn LED off


  // Connect to WiFi

  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {

    delay(1000);

    Serial.println("Connecting to WiFi...");

  }

  Serial.println("Connected!");


  // Define web server routes

  server.on("/", handleRoot);

  server.on("/setColor", handleSetColor);

  server.begin();

  Serial.println("Web server started!");

  Serial.print("IP Address: ");

  Serial.println(WiFi.localIP());

}


void loop() {

  server.handleClient();

}
